[题目传送门](https://www.luogu.com.cn/problem/P4135)

<!-- more -->

可以发现这个题居然不卡空间。于是可以随便乱搞。

强制在线，考虑暴力分块，设块数为 $t$。

设 $cnt_{i,j}$ 表示从第 $i$ 块到第 $t$ 块，$j$ 的出现次数。$sum_{i,j}$ 表示从第 $i$ 块到第 $j$ 块的答案，显然预处理可以做到 $O(nt)$。

这两个数组大小差不多要开到 4e7，但是这题不卡空间，无所谓了。

然后利用类似差分的思想可以 $O(1)$ 求任意两个块间某个数出现的次数。整块查表散块暴力即可。注意某个数第一次出现对答案没有贡献。

这样可以做到 $O(nt+\dfrac{mn}{t})$，$n,m$ 同阶，取 $t=\sqrt{n}$，于是可以做到 $O(n \sqrt{n})$。

代码：

```cpp
#include<bits/stdc++.h>
#include<iostream>
#define ll long long
#define back return 
#define ri register int 
using namespace std;
ll read()
{
	ll x=0,f=1;
	char ch=getchar();
	while(ch<'0'||ch>'9')
	{
		if(ch=='-')
			f=-1;
		ch=getchar();
	}
	while(ch>='0'&&ch<='9')
	{
		x=x*10+ch-'0';
		ch=getchar();
	}
	back x*f;
}
bool vis[100005];
int n,m,c,t,len,a[100005],pos[100005],L[100005],R[100005]; 
int cnt[330][100005],sum[1505][1505];
int jsq[100005];
ll ask(ll l,ll r)
{
	int ans=0,p1=pos[l],q1=pos[r];
	if(p1==q1)
	{
		for(ri i=l;i<=r;i++)
		{
			jsq[a[i]]++;
			if(jsq[a[i]]%2&&jsq[a[i]]!=1)
				ans--;
			if(jsq[a[i]]%2==0)
				ans++;
		}
		for(ri i=l;i<=r;i++)
			jsq[a[i]]=0;
		back ans;	
	}
	ans=sum[p1+1][q1-1];
	for(ri i=l;i<=R[p1];i++)
	{
		jsq[a[i]]++;
		if((jsq[a[i]]+cnt[p1+1][a[i]]-cnt[q1][a[i]])%2&&(jsq[a[i]]+cnt[p1+1][a[i]]-cnt[q1][a[i]])!=1)
			ans--;
		if((jsq[a[i]]+cnt[p1+1][a[i]]-cnt[q1][a[i]])%2==0)
			ans++;
	}
	for(ri i=L[q1];i<=r;i++)
	{
		jsq[a[i]]++;
		if((jsq[a[i]]+cnt[p1+1][a[i]]-cnt[q1][a[i]])%2&&(jsq[a[i]]+cnt[p1+1][a[i]]-cnt[q1][a[i]])!=1)
			ans--;
		if((jsq[a[i]]+cnt[p1+1][a[i]]-cnt[q1][a[i]])%2==0)
			ans++;
	}
	for(ri i=l;i<=R[p1];i++)
		jsq[a[i]]=0;
	for(ri i=L[q1];i<=r;i++)
		jsq[a[i]]=0;
	back ans;	
}
int main()
{
	n=read(),c=read(),m=read();
	for(ri i=1;i<=n;i++)
		a[i]=read();
	t=sqrt(n),len=n/t;
	for(ri i=1;i<=t;i++)
		L[i]=(i-1)*len+1,R[i]=i*len;
	if(R[t]<n)
		t++,L[t]=R[t-1]+1,R[t]=n;
	for(ri i=1;i<=t;i++)
		for(ri j=L[i];j<=R[i];j++)
			pos[j]=i;
	for(ri i=1;i<=t;i++)
		for(ri j=i;j<=t;j++)
			for(ri k=L[j];k<=R[j];k++)
				cnt[i][a[k]]++;
	for(ri i=1;i<=t;i++)
	{
		memset(vis,0,sizeof(vis));
		memset(jsq,0,sizeof(jsq));
		for(ri j=i;j<=t;j++)
		{
			sum[i][j]=sum[i][j-1];
			for(ri k=L[j];k<=R[j];k++)
			{
				jsq[a[k]]++;
				if(jsq[a[k]]%2&&jsq[a[k]]!=1)
					sum[i][j]--;
				if(jsq[a[k]]%2==0)
					sum[i][j]++;
			}
		}		
	}
	ll x=0;
	memset(jsq,0,sizeof(jsq));
	while(m--)
	{
		ll l=read(),r=read();
		l=(l+x)%n+1,r=(r+x)%n+1;
		if(l>r)
			swap(l,r);
		x=ask(l,r);
		cout<<x<<"\n";
	}
    back 0;
}
```
