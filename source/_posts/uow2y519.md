[题目传送门](https://www.luogu.com.cn/problem/P2476)

<!-- more -->

这个题有点牛逼啊。

首先为了方便计数，我们不关心某种颜色具体被用了多少次，只关心能用 $i$ 次的颜料都有几种。发现每种颜色使用最多不超过五次，可以直接考虑设置 $dp_{a,b,c,d,e}$ 表示能用 $1$ 次的颜色有 $a$ 种，能用 $2$ 次的有 $b$ 种……以此类推。

但是这样没法转移，所以还要记录一维 $last$ 表示上一种用的是什么颜色，这样计数时就不会统计到非法状态。

注意统计方案数的时候要考虑上一种用的是什么颜色，这种颜色在当前统计中不能使用，应该处理为可使用的颜色数 $-1$。

然后利用 $dp$ 数组顺带记搜一下就随便过了。

```cpp
#include<bits/stdc++.h>
#define ll long long
#define back return
#define ri register int
#define ull unsigned ll
using namespace std;
ll read()
{
	ll x=0,f=1;
	char ch=getchar();
	while(ch<'0'||ch>'9')
	{
		if(ch=='-')
			f=-1;
		ch=getchar();
	}
	while(ch>='0'&&ch<='9')
	{
		x=x*10+ch-'0';
		ch=getchar();
	}
	back x*f;
}
const int mod=1e9+7;
ll k,cnt[20],dp[17][17][17][17][17][7];
ll dfs(int a,int b,int c,int d,int e,int lst)
{
	if(dp[a][b][c][d][e][lst])
		back dp[a][b][c][d][e][lst]; 
	if(a+b+c+d+e==0)
		back 1;
	ll res=0;
	if(a)
		res=(res+(a-(lst==2))*dfs(a-1,b,c,d,e,1))%mod;
	if(b)
		res=(res+(b-(lst==3))*dfs(a+1,b-1,c,d,e,2))%mod;
	if(c)
		res=(res+(c-(lst==4))*dfs(a,b+1,c-1,d,e,3))%mod;
	if(d)
		res=(res+(d-(lst==5))*dfs(a,b,c+1,d-1,e,4))%mod;
	if(e)
		res=(res+e*dfs(a,b,c,d+1,e-1,5))%mod;
	dp[a][b][c][d][e][lst]=res%mod;
	back dp[a][b][c][d][e][lst]%mod;
}
int main()
{
	k=read();
	for(ri i=1;i<=k;i++)
	{
		int c=read();
		cnt[c]++;
	}
	cout<<dfs(cnt[1],cnt[2],cnt[3],cnt[4],cnt[5],0)<<"\n";
	back 0;
}
```
