[题目传送门](https://www.luogu.com.cn/problem/P1462)

萌萌题。

一眼二分答案，设 $f_i$ 表示经过 $i$ 点的最短路上的最大点权，求最短路的时候入队 $i$ 点时顺便更新一下$f_i$ 即可，注意点权超过当前答案的点不能走。如果最后最短路长度依然 $>b$  说明当前答案无解，多次二分注意清空。

然后注意无解输出 $AFK$ 而不是 $-1$。

时间复杂度 $O(m \log n \log f_i)$。

代码：

```cpp
#include<bits/stdc++.h>
#define ll long long
#define ri register int
#define back return
using namespace std;
ll read()
{
	ll x=0,f=1;
	char ch=getchar();
	while(ch<'0'||ch>'9')
	{
		if(ch=='-')
			f=-1;
		ch=getchar();
	}
	while(ch>='0'&&ch<='9')
	{
		x=x*10+ch-'0';
		ch=getchar();
	}
	back x*f;
}
ll n,m,b,p[50005],f[50005],dis[50005],l,r,ans=1e17; 
vector<ll> a[50005],w[50005];
bool vis[50005];
struct node
{
	ll x,dis;
	node(ll x_,ll dis_)
	{
		x=x_;
		dis=dis_;
	}
	friend bool operator < (node a,node b)
	{
		back a.dis>b.dis;
	}
};
priority_queue<node> q;
void dijkstra(int s,int max1)
{
	q.push(node(s,0)),dis[s]=0;
	while(!q.empty())
	{
		int now=q.top().x;
		q.pop();
		if(vis[now]||f[now]>max1)
			continue;
		vis[now]=1;
		for(ri i=0;i<a[now].size();i++)
		{
			int to=a[now][i],val=w[now][i];
			if(dis[to]>dis[now]+val)
			{
				dis[to]=dis[now]+val;
				f[to]=max(f[to],f[now]);
				q.push(node(to,dis[to]));
			}		
		}
	}
}
ll check(ll ans)
{
	for(ri i=1;i<=n;i++)
		dis[i]=1e17,vis[i]=0,f[i]=p[i];
	dijkstra(1,ans);
	if(dis[n]>b)
		back -1;
	back f[n];
}
int main()
{
   	n=read(),m=read(),b=read();
   	for(ri i=1;i<=n;i++)	
   		p[i]=read(),f[i]=p[i],r=max(r,p[i]); 
   	for(ri i=1;i<=m;i++)
   	{
   		int x=read(),y=read(),w1=read();
   		a[x].push_back(y),a[y].push_back(x);
   		w[x].push_back(w1),w[y].push_back(w1);
   	}
   	int mid=(l+r)/2;
   	while(l<=r)
   	{
   		mid=(l+r)/2;
   		ll res=check(mid);
   		if(res==-1)
   			l=mid+1;
   		else
   			ans=min(ans,res),r=mid-1;
   	}
   	if(ans==1e17)
   		cout<<"AFK"<<"\n";
   	else
   		cout<<ans<<"\n";
    back 0;
}
```
